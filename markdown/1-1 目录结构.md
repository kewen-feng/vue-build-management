# 目录结构说明

|-- dist // build打包的生产文件
|-- node_modules // 依赖模块
|-- src // 源代码
|  |-- api // api集合
|  |-- assets // 静态资源
|  |-- components // 公共组件
|  |-- router // 路由
|  |-- store // 全局store管理
|  |-- utils // 全局公用方法
|  |-- views // 视图
|  |-- App.vue // 入口页面
|  |-- main.js // 入口 加载路径 初始化
|-- babel.config.js // Babel配置
|-- package.json // 项目清单，存储所有已安装软件包的名称和版本的地方
